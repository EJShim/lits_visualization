cmake_minimum_required(VERSION 3.0)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_INCLUDE_CURRENT_DIR ON)

set(SRC_FILES 
    main
    resources/resources.qrc
    gui/E_Window.cpp
    manager/E_Manager.cpp
    manager/E_VolumeManager.cpp
)

set(PUBLISH FALSE)

include_directories(
    gui/    
    manager/
)
# add_executable(run ${SRC_FILES})
if(${PUBLISH})
    if (WIN32)
        add_executable(run WIN32 ${SRC_FILES})
    endif (WIN32)
    if (APPLE)
        set(MACOSX_BUNDLE_ICON_FILE pantone-2.icns)
        set(myApp_ICON ${CMAKE_CURRENT_SOURCE_DIR}/resources/icons/pantone-2.icns)
        set_source_files_properties(${myApp_ICON} PROPERTIES MACOSX_PACKAGE_LOCATION "Resources")
        add_executable(run MACOSX_BUNDLE ${myApp_ICON} ${SRC_FILES})
    endif (APPLE)
else(${PUBLISH})
    add_executable(run ${myApp_ICON} ${SRC_FILES})
endif(${PUBLISH})


#VTK Configuration - contains QT
find_package(VTK REQUIRED)
include(${VTK_USE_FILE})
target_link_libraries(run ${VTK_LIBRARIES})

#ITK Confiugration
find_package(ITK REQUIRED)
include(${ITK_USE_FILE})
target_link_libraries(run ${ITK_LIBRARIES})
